## 5.6 圖表生成程式

### Python 圖表生成腳本

```python
# 檔案位置: docs/試營運報告/圖表/generate_charts.py

import matplotlib.pyplot as plt
import pandas as pd
import numpy as np

# 設定中文字體
plt.rcParams['font.sans-serif'] = ['Microsoft JhengHei']
plt.rcParams['axes.unicode_minus'] = False

# 1. 服務量趨勢圖
def generate_service_trend():
    months = ['第1月', '第2月', '第3月']
    physical = [50, 40, 35]
    voice = [10, 30, 45]

    plt.figure(figsize=(10, 6))
    plt.plot(months, physical, marker='o', label='實體訪視')
    plt.plot(months, voice, marker='s', label='語音關懷')
    plt.title('每月服務量趨勢')
    plt.xlabel('月份')
    plt.ylabel('服務次數')
    plt.legend()
    plt.grid(True, alpha=0.3)
    plt.savefig('服務量趨勢.png', dpi=300, bbox_inches='tight')
    plt.close()

# 2. 碳排放對比圖
def generate_carbon_comparison():
    categories = ['導入前', '導入後']
    values = [150, 80]

    plt.figure(figsize=(8, 6))
    bars = plt.bar(categories, values, color=['#ff6b6b', '#51cf66'])
    plt.title('月平均碳排放對比')
    plt.ylabel('碳排放量 (kg CO₂e)')

    for bar in bars:
        height = bar.get_height()
        plt.text(bar.get_x() + bar.get_width()/2., height,
                f'{height} kg', ha='center', va='bottom')

    plt.savefig('碳排放對比.png', dpi=300, bbox_inches='tight')
    plt.close()

# 3. 成本節省圖
def generate_cost_saving():
    categories = ['人力成本', '交通成本', '總成本']
    before = [80000, 15000, 95000]
    after = [60000, 8000, 70000]

    x = np.arange(len(categories))
    width = 0.35

    plt.figure(figsize=(10, 6))
    plt.bar(x - width/2, before, width, label='導入前', color='#ff6b6b')
    plt.bar(x + width/2, after, width, label='導入後', color='#51cf66')

    plt.title('成本對比分析')
    plt.ylabel('金額 (元)')
    plt.xticks(x, categories)
    plt.legend()
    plt.grid(True, alpha=0.3, axis='y')
    plt.savefig('成本對比.png', dpi=300, bbox_inches='tight')
    plt.close()

if __name__ == '__main__':
    generate_service_trend()
    generate_carbon_comparison()
    generate_cost_saving()
    print('圖表生成完成！')
```
